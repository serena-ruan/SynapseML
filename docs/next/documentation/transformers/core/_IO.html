HTTPTransformer#

  • Python
  • Scala
from synapse.ml.io.http import *from pyspark.sql.functions import udf, colfrom requests import Request
def world_bank_request(country):    return Request("GET", "http://api.worldbank.org/v2/country/{}?format=json".format(country))
df = (spark.createDataFrame([("br",), ("usa",)], ["country"])      .withColumn("request", http_udf(world_bank_request)(col("country"))))
ht = (HTTPTransformer()      .setConcurrency(3)      .setInputCol("request")      .setOutputCol("response"))
display(ht.transform(df))
import com.microsoft.azure.synapse.ml.io.http._
val ht = (new HTTPTransformer()      .setConcurrency(3)      .setInputCol("request")      .setOutputCol("response"))

Featurize#

  • Python
  • Scala
from synapse.ml.featurize import *
dataset = spark.createDataFrame([    (0, 2, 0.50, 0.60, "pokemon are everywhere"),    (1, 3, 0.40, 0.50, "they are in the woods"),    (0, 4, 0.78, 0.99, "they are in the water"),    (1, 5, 0.12, 0.34, "they are in the fields"),    (0, 3, 0.78, 0.99, "pokemon - gotta catch em all")], ["Label", "col1", "col2", "col3"])
featureColumns = filter(lambda x: x != "Label", dataset.columns)
feat = (Featurize()      .setNumFeatures(10)      .setOutputCol("testColumn")      .setInputCols(list(featureColumns))      .setOneHotEncodeCategoricals(False))
display(feat.fit(dataset).transform(dataset))
import com.microsoft.azure.synapse.ml.featurize._import spark.implicits._
val dataset = Seq(      (0, 2, 0.50, 0.60, "pokemon are everywhere"),      (1, 3, 0.40, 0.50, "they are in the woods"),      (0, 4, 0.78, 0.99, "they are in the water"),      (1, 5, 0.12, 0.34, "they are in the fields"),      (0, 3, 0.78, 0.99, "pokemon - gotta catch em all")).toDF("Label", "col1", "col2", "col3")
val featureColumns = dataset.columns.filter(_ != "Label")
val feat = (new Featurize()      .setNumFeatures(10)      .setOutputCol("testColumn")      .setInputCols(featureColumns)      .setOneHotEncodeCategoricals(false))
display(feat.fit(dataset).transform(dataset))
